Module pl.Set

A Set class.

 > Set = require 'pl.Set'
 > = Set{'one','two'} == Set{'two','one'}
 true
 > fruit = Set{'apple','banana','orange'}
 > = fruit['banana']
 true
 > = fruit['hazelnut']
 nil
 > colours = Set{'red','orange','green','blue'}
 > = fruit,colours
 [apple,orange,banana]   [blue,green,orange,red]
 > = fruit+colours
 [blue,green,apple,red,orange,banana]
 > = fruit*colours
 [orange]
